create database if not exists movie_theater_ms;
create schema if not exists movie_theater_ms;

create table if not exists theaters(
	theater_id SERIAL primary key,
	name VARCHAR(50) not null,
	address VARCHAR(100) not null,
	rating DECIMAL(2,1) default 0
);

create table if not exists genres(
	genre_id SERIAL primary key,
	name VARCHAR(30) not null
);

create table if not exists films(
	film_id SERIAL primary key,
	name VARCHAR(100) not null,
	duration TIMESTAMP not null,
	genre_id references genres(genre_id),
	release_date DATE not null CHECK(release_date > '01.01.2026'),
	age_restriction VARCHAR(3) not null,
	rating DECIMAL(2,1) not null default 0,
	description TEXT not null
);

create table if not exists staff(
	staff_id SERIAL primary key,
	name VARCHAR(50) not null,
	phone_number VARCHAR(12) not null,
	age INT not null,
	job_position VARCHAR(50) not null,
	shift VARCHAR(20) not null,
	salary VARCHAR(10) not null,
	theater_id INT not null references theaters(theater_id) 
);

create table if not exists customers(
	customer_id SERIAL primary key,
	name VARCHAR(50) not null,
	phone_number VARCHAR(12) not null,
	credt_card VARCHAR(12)
);

create type purchase_option as ENUM('Cash', 'Credit_card');

create table if not exists operations(
	operation_id SERIAL primary key,
	purchase_type purchase_option not null,
	date DATE not null check(date > '01.012026'),
	time TIMESTAMP not null
);

create table if not exists seat_combo(
	row_and_number VARCHAR(5) not null unique
);

create type ticket_option as ENUM('Child','Adult','Student');

create table if not exists tickets(
	ticket_id SERIAL primary key,
	ticket_type ticket_option not null,
	price VARCHAR(10) not null
);

create type seat_option as ENUM('Normal','Comfort','VIP');

create table if not exists seats(
	seat_id SERIAL primary key,
	seat_type seat_option not null,
	number INT not null,
	row VARCHAR(3) not null,
	price VARCHAR(10) not null default 800
);

create table if not exists schedule(
	schedule_id SERIAL primary key,
	theater_id INT not null references theaters(theater_id),
	film_id INT not null references films(film_id),
	date DATE not null check(date > '01.01.2026'),
	time TIMESTAMP not null
);

create table if not exists screenings(
	screening_id SERIAL primary key,
	film_id INT not null references films(film_id),
	theater_id INT not null references theaters(theater_id),
	screening_date DATE not null check (screening_date > '01.01.2026'),
	start_time TIMESTAMP not null,
	end_time TIMESTAMP not null
);

create table if not exists reservations(
	reservation_id SERIAL primary key,
	customer_id INT not null references customers(customer_id),
	seat_id INT not null references seats(seat_id),
	ticket_id INT not null references tickets(ticket_id),
	film_id INT not null references films(film_id),
	operation_id INT not null references operations(opearion_id),
	total_price VARCHAR(10) not null GENERATE as (price * )
);
